\subsection{数学公式}
值得注意的是，微分符号$d$应该用直体，写作$\d x$，$max$应写作$\max$, $min$应写作$\min$， $sin$应写作$\sin$。

\begin{definition}[勾股定理]
	假设直角三角形的三边长为$a$、 $b$、 $c$，斜边长为$c$，则
	$$a^2+b^2=c^2.$$
\end{definition}

\begin{lemma}[勾股定理]
	假设直角三角形的三边长为$a$、 $b$、 $c$，斜边长为$c$，则
	$$a^2+b^2=c^2.$$
\end{lemma}
\begin{theorem}[勾股定理]
	假设直角三角形的三边长为$a$、 $b$、 $c$，斜边长为$c$，则
	$$a^2+b^2=c^2.$$
\end{theorem}
